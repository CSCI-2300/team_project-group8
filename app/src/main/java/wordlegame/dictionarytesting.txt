public class WordGenerator
{
    private ArrayList<String> animalWords;
    private ArrayList<String> countryWords;
    private ArrayList<String> cityWords;

    public WordGenerator()
    {
        animalWords = new ArrayList<>();
        countryWords = new ArrayList<>();
        cityWords = new ArrayList<>();

        animalWords.add("horse");
        animalWords.add("zebra");
        animalWords.add("llama");

        countryWords.add("japan");
        countryWords.add("china");
        countryWords.add("france");
        countryWords.add("italy");
        countryWords.add("spain");

        cityWords.add("paris");
        cityWords.add("tokyo");
     
    }

    public String generateWord(String category)
    {
        ArrayList<String> wordList = new ArrayList<>();
        switch(category)
        {
            case "Animals":
                wordList = animalWords;
                break;
            case "Countries":
                wordList = countryWords;
                break;
            case "Cities":
                wordList = cityWords;
                break;
            default:
                System.out.println("Invalid category");
                break;
        }
////////////you dont need to most/ any code below 
        String selectedWord = "";

        //if we end up using a dictionary we can use this method
        while(selectedWord.length() != 5)
        {
            selectedWord = wordList.get((int)(Math.random()*wordList.size()));
        }

         //if we end up using a dictionary we can use this method
        char[] wordAsCharArray = selectedWord.toCharArray();
        boolean valid = true;
        for(int i = 0; i < wordAsCharArray.length; i++)
        {
            int count = 0;
            for(int j = 0; j < wordAsCharArray.length; j++)
            {
                if(wordAsCharArray[i] == wordAsCharArray[j])
                {
                    count++;
                }
            }
            if(count > 3)
            {
                valid = false;
            }
        }

        /////change word to uppercase , this will be more helpful 
        if(!valid)
        {
            return generateWord(category);
        }
        else
        {
            return selectedWord.toUpperCase();
        }
    }
}
